---
tweet: Using onCreatePage in @gatsbyjs allows you to change various aspects of pages created by plugins you don't control, such as the URL.
title: Change Page URL
---

```js
exports.onCreatePage = ({ page, actions }) => {
  const { createPage, deletePage } = actions;
  const oldPage = Object.assign({}, page);
  if (oldPage.path === "/posts") {
    page.path = "/blog-posts";
    deletePage(oldPage);
    createPage(page);
  }
};
```
