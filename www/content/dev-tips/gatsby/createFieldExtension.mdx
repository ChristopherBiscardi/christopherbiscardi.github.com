---
tweet: Field Extensions in @gatsbyjs allow you to extend resolvers for specific fields with additional functionality via GraphQL directives
title: createFieldExtension
---

```js
exports.createSchemaCustomization = ({ actions }) => {
  const { createTypes, createFieldExtension } = actions;

  createFieldExtension({
    name: "shout",
    extend: () => ({
      resolve(source, args, context, info) {
        return String(source[info.fieldName]).toUpperCase();
      }
    })
  });

  const typeDefs = `
    type BlogPost {
      title: String! @shout
    }
  `;
  createTypes(typeDefs);
};
```
