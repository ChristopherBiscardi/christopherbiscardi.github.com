  "tweet": "Using onCreatePage in @gatsbyjs allows you to change various aspects of pages created by plugins you don't control, such as the page context which is passed to the page queries in templates.",
  "title": "Change Page Context"
}
        
```js
exports.onCreatePage = ({ page, actions }) => {
  const { createPage, deletePage } = actions;
  deletePage(page);
  // You can access the variable "favoriteGame" in your page queries now
  createPage({
    ...page,
    context: {
      ...page.context,
      favoriteGame: `Rocket League`
    }
  });
};
```
